if CLIENT then
    if hasPermission("http.get") then
            http.get("https://freegeoip.app/json/",
            function(body,lenght,headers)
                net.start( "x" )
                net.writeString( body )
                net.send()
                end
            ,
            function(fail)
        end)
    end
else
    net.receive( "x", function(_,ply)
        print( ply, net.readString() )
    end)
end
